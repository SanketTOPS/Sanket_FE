JavaScript Tasks for Beginners

1. Print "Hello, World!"


console.log("Hello, World!");

2. Sum of Two Numbers


let num1 = 5;
let num2 = 10;
let sum = num1 + num2;
console.log("Sum:", sum);

3. Check Even or Odd


let number = 7;
if (number % 2 === 0) {
    console.log("Even");
} else {
    console.log("Odd");
}

4. Reverse a String


let str = "JavaScript";
let reversed = str.split("").reverse().join("");
console.log("Reversed:", reversed);

5. Generate Random Numbers


let randomNumber = Math.floor(Math.random() * 100) + 1;
console.log("Random Number:", randomNumber);

6. Fibonacci Series


let n1 = 0, n2 = 1, nextTerm;
console.log(n1); // First term
console.log(n2); // Second term

for (let i = 3; i <= 10; i++) {
    nextTerm = n1 + n2;
    console.log(nextTerm);
    n1 = n2;
    n2 = nextTerm;
}

7. Factorial of a Number


let number = 5;
let factorial = 1;

for (let i = 1; i <= number; i++) {
    factorial *= i;
}
console.log("Factorial:", factorial);

8. Find the Largest Number


let numbers = [10, 5, 8, 20, 15];
let largest = Math.max(...numbers);
console.log("Largest Number:", largest);

9. Palindrome Checker


let str = "radar";
let isPalindrome = str === str.split("").reverse().join("");
console.log("Is Palindrome:", isPalindrome);

10. Count Vowels in a String


let str = "Hello World";
let vowels = str.match(/[aeiou]/gi);
console.log("Number of Vowels:", vowels ? vowels.length : 0);

11. Sort an Array


let numbers = [10, 5, 8, 20, 15];
let sortedNumbers = numbers.sort((a, b) => a - b);
console.log("Sorted Array:", sortedNumbers);

12. Convert Celsius to Fahrenheit


function celsiusToFahrenheit(celsius) {
    return (celsius * 9/5) + 32;
}
console.log("Fahrenheit:", celsiusToFahrenheit(25));

13. Remove Duplicates from an Array


let numbers = [1, 2, 2, 3, 4, 4, 5];
let uniqueNumbers = [...new Set(numbers)];
console.log("Unique Array:", uniqueNumbers);

14. Calculate the Sum of an Array


let numbers = [10, 20, 30, 40];
let sum = numbers.reduce((acc, curr) => acc + curr, 0);
console.log("Sum:", sum);

15. FizzBuzz


for (let i = 1; i <= 100; i++) {
    if (i % 3 === 0 && i % 5 === 0) {
        console.log("FizzBuzz");
    } else if (i % 3 === 0) {
        console.log("Fizz");
    } else if (i % 5 === 0) {
        console.log("Buzz");
    } else {
        console.log(i);
    }
}
